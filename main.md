# 侧边显示器方案

### 连接方式

```javascript
// 示例代码：Node.js 检测显示器连接
const { exec } = require('child_process');

// 检测外部显示器是否连接
function checkExternalDisplay() {
  return new Promise((resolve, reject) => {
    exec('xrandr --query', (error, stdout, stderr) => {
      if (error) {
        reject(error);
        return;
      }

      // 查找连接的显示器
      const displays = stdout.match(/(\w+-\d+)\s+connected/g);
      const externalDisplays = displays ? displays.filter(d => !d.includes('eDP')) : [];

      resolve(externalDisplays.length > 0);
    });
  });
}

// 主函数
async function main() {
  try {
    const isExternalDisplayConnected = await checkExternalDisplay();
    console.log('外部显示器连接状态:', isExternalDisplayConnected);

    if (isExternalDisplayConnected) {
      console.log('建议使用扩展模式以获得最佳体验');
    } else {
      console.log('未检测到外部显示器');
    }
  } catch (error) {
    console.error('检测显示器连接失败:', error);
  }
}

// 运行程序
main();
```
